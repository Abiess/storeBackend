# Cronjob für automatische Datenbank-Backups
# Installation: crontab -e
# Dann diese Zeile hinzufügen:

# Tägliches Backup um 2:00 Uhr nachts
0 2 * * * /opt/storebackend/scripts/backup-database.sh >> /var/log/storebackend/backup.log 2>&1

# Wöchentliche Health Checks (jeden Montag um 9:00 Uhr)
0 9 * * 1 /opt/storebackend/scripts/health-check.sh >> /var/log/storebackend/health-check.log 2>&1

# Alte Logs aufräumen (älter als 30 Tage, täglich um 3:00 Uhr)
0 3 * * * find /var/log/storebackend -name "*.log" -mtime +30 -delete

# System Updates (jeden Sonntag um 4:00 Uhr)
0 4 * * 0 apt update && apt upgrade -y >> /var/log/apt/auto-upgrade.log 2>&1

# SSL Zertifikat erneuern (täglich um 3:30 Uhr - Certbot prüft automatisch)
30 3 * * * certbot renew --quiet
